<?php
include "config.php";
session_start();


if (!isset($_SESSION['user_id'])) {
    header("Location: index.php");
    exit();
}

$user_id = $_SESSION['user_id'];


if (isset($_POST['create_post'])) {
    $content = trim($_POST['post_content']);
    $imageName = "";

    if (!empty($_FILES['post_image']['name'])) {
        $targetDir = "uploads/posts/";
        if (!is_dir($targetDir)) {
            mkdir($targetDir, 0777, true);
        }
        $imageName = time() . "_" . $_FILES['post_image']['name'];
        move_uploaded_file($_FILES['post_image']['tmp_name'], $targetDir . $imageName);
    }

    if ($content != "" || $imageName != "") {
        $sql = "INSERT INTO posts (user_id, content, image, like_count) VALUES ($user_id, '$content', '$imageName', 0)";
        $conn->query($sql);
    }
}

if (isset($_POST['like_post'])) {
    $post_id = $_POST['post_id'];
    $conn->query("UPDATE posts SET like_count = like_count + 1 WHERE id = $post_id");
}


if (isset($_POST['comment_post'])) {
    $post_id = $_POST['post_id'];
    $comment = trim($_POST['comment']);
    $username = $_SESSION['username'];

    if ($comment != "") {
        $conn->query("INSERT INTO comments (post_id, user_name, comment) VALUES ($post_id, '$username', '$comment')");
    }
}


$posts = $conn->query("
    SELECT posts.*, users.username, users.profile_pic 
    FROM posts 
    JOIN users ON posts.user_id = users.id 
    ORDER BY posts.created_at DESC
");
